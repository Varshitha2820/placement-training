        System.out.print("Enter the elements of the array:");
        for(int i=0;i<n;i++){
            arr[i]=scanner.nextInt();
        }
        int key=7,s=0,e=arr.length-1,c=0;
            while (s <= e) {
                int m = (s + e) / 2;
                if (arr[m] == key) {
                    c++; 
                    break;
                } 
                else if (arr[m] < key) s = m + 1; 
                else  e = m - 1;

}
